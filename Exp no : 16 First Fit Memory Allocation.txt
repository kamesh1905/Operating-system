#include <stdio.h>

OS-Experiment-16 

int main() {
    int nb, np;
    int block[10], process[10], allocation[10];

    printf("Enter number of blocks: ");
    scanf("%d", &nb);

    printf("Enter size of each block:\n");
    for(int i = 0; i < nb; i++)
        scanf("%d", &block[i]);

    printf("Enter number of processes: ");
    scanf("%d", &np);

    printf("Enter size of each process:\n");
    for(int i = 0; i < np; i++)
        scanf("%d", &process[i]);

    for(int i = 0; i < np; i++)
        allocation[i] = -1;

    for(int i = 0; i < np; i++) {
        for(int j = 0; j < nb; j++) {
            if(block[j] >= process[i]) {
                allocation[i] = j;
                block[j] -= process[i];
                break;
            }
        }
    }

    printf("\nProcess No\tBlock No\n");
    for(int i = 0; i < np; i++) {
        if(allocation[i] != -1)
            printf("%d\t\t%d\n", i+1, allocation[i]+1);
        else
            printf("%d\t\tNot Allocated\n", i+1);
    }

    return 0;
}
