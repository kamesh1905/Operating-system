#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, head, disk_size, total = 0;
    int req[20], temp;

    printf("Enter number of requests: ");
    scanf("%d", &n);

    printf("Enter request queue:\n");
    for(int i = 0; i < n; i++)
        scanf("%d", &req[i]);

    printf("Enter initial head position: ");
    scanf("%d", &head);

    printf("Enter disk size: ");
    scanf("%d", &disk_size);

    for(int i = 0; i < n-1; i++) {
        for(int j = i+1; j < n; j++) {
            if(req[i] > req[j]) {
                temp = req[i];
                req[i] = req[j];
                req[j] = temp;
            }
        }
    }

    int pos;
    for(int i = 0; i < n; i++) {
        if(req[i] > head) {
            pos = i;
            break;
        }
    }

    for(int i = pos; i < n; i++) {
        total += abs(req[i] - head);
        head = req[i];
    }

    total += abs((disk_size - 1) - head);
    head = disk_size - 1;

    for(int i = pos-1; i >= 0; i--) {
        total += abs(req[i] - head);
        head = req[i];
    }

    printf("Total seek time = %d\n", total);
    return 0;
}
