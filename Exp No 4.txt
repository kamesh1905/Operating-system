Experiment 4-CPU Scheduling-SJF

#include <stdio.h>

int main() {
    int n, i, j, temp;
    float avgWT = 0, avgTAT = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    int bt[n], wt[n], tat[n], pid[n];

    for (i = 0; i < n; i++) {
        pid[i] = i + 1;
        printf("Enter burst time of P%d: ", pid[i]);
        scanf("%d", &bt[i]);
    }

    // Sort by burst time (SJF)
    for (i = 0; i < n - 1; i++) {
        for (j = i + 1; j < n; j++) {
            if (bt[i] > bt[j]) {
                temp = bt[i]; bt[i] = bt[j]; bt[j] = temp;
                temp = pid[i]; pid[i] = pid[j]; pid[j] = temp;
            }
        }
    }

    wt[0] = 0;
    for (i = 1; i < n; i++)
        wt[i] = wt[i - 1] + bt[i - 1];

    for (i = 0; i < n; i++) {
        tat[i] = wt[i] + bt[i];
        avgWT += wt[i];
        avgTAT += tat[i];
    }

    printf("\nProcess\tBT\tWT\tTAT\n");
    for (i = 0; i < n; i++)
        printf("P%d\t%d\t%d\t%d\n", pid[i], bt[i], wt[i], tat[i]);

    printf("\nAverage WT = %.2f", avgWT / n);
    printf("\nAverage TAT = %.2f\n", avgTAT / n);

    return 0;
}
