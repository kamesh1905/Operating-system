#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, head, total = 0;
    int req[20];

    printf("Enter number of requests: ");
    scanf("%d", &n);

    printf("Enter request queue:\n");
    for(int i = 0; i < n; i++)
        scanf("%d", &req[i]);

    printf("Enter initial head position: ");
    scanf("%d", &head);

    for(int i = 0; i < n; i++) {
        total += abs(req[i] - head);
        head = req[i];
    }

    printf("Total seek time = %d\n", total);
    return 0;
}
