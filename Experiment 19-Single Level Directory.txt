#include <stdio.h>
#include <string.h>

int main() {
    char files[10][20], name[20];
    int n = 0, choice, found;

    while(1) {
        printf("\n1.Create  2.Delete  3.Search  4.Display  5.Exit\n");
        scanf("%d", &choice);

        if(choice == 1) {
            printf("Enter file name: ");
            scanf("%s", files[n++]);
        }
        else if(choice == 2) {
            printf("Enter file name to delete: ");
            scanf("%s", name);
            for(int i = 0; i < n; i++) {
                if(strcmp(files[i], name) == 0) {
                    strcpy(files[i], files[n-1]);
                    n--;
                }
            }
        }
        else if(choice == 3) {
            found = 0;
            printf("Enter file name to search: ");
            scanf("%s", name);
            for(int i = 0; i < n; i++) {
                if(strcmp(files[i], name) == 0)
                    found = 1;
            }
            if(found)
                printf("File found\n");
            else
                printf("File not found\n");
        }
        else if(choice == 4) {
            for(int i = 0; i < n; i++)
                printf("%s\n", files[i]);
        }
        else
            break;
    }
    return 0;
}
