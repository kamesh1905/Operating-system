#include <stdio.h>
#include <pthread.h>

void* printNumbers(void* arg) {
    for(int i = 1; i <= 5; i++) {
        printf("Number: %d\n", i);
    }
    return NULL;
}

void* printLetters(void* arg) {
    for(char c = 'A'; c <= 'E'; c++) {
        printf("Letter: %c\n", c);
    }
    return NULL;
}

int main() {
    pthread_t t1, t2;

    pthread_create(&t1, NULL, printNumbers, NULL);
    pthread_create(&t2, NULL, printLetters, NULL);

    pthread_join(t1, NULL);
    pthread_join(t2, NULL);

    return 0;
}
