#include <stdio.h>
#include <string.h>

int main() {
    FILE *fp;
    char line[200];
    char word[50];

    fp = fopen("example.txt", "r");
    if (fp == NULL) {
        printf("File not found\n");
        return 1;
    }

    printf("Enter word to search: ");
    scanf("%s", word);

    while (fgets(line, sizeof(line), fp)) {
        if (strstr(line, word))
            printf("%s", line);
    }

    fclose(fp);
    return 0;
}
