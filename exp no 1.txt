Experiment-1 Program to get process information

#include <stdio.h>
#include <unistd.h>
int main() {
    pid_t pid;
    pid=fork();
    if (pid == 0) {
        printf("Child Process: PID = %d, PPID = %d\n",
               getpid(), getppid());
    }
    else if (pid > 0) {
        printf("Parent Process: PID = %d, PPID = %d\n",
               getpid(), getppid());
    }
    else {
        printf("Fork failed\n");
    }

    return 0;
}
